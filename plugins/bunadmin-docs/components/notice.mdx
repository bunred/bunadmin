# Notice
### A notification prompt will pop up, and the message will be stored locally

<Button color="secondary" onClick={async ()=> {
    await notice({
      title: "Notice done",
      content: "This is a Notice test"
    })
}}>Initiate a success Notice</Button>

``` ts
import { notice } from "@bunred/bunadmin"
import { Button } from "@material-ui/core"
//...
    await notice({

    })
```

``` tsx
import { notice } from "@bunred/bunadmin"

```

## Props
``` ts
interface Props {
  title: string
  severity?: "success" | "info" | "warning" | "error" | null
  content?: string | object
}
```
