``` ts
import { MultipleSelector } from "@bunred/bunadmin"
//...
<MultipleSelector
 columnDef={{ lookup: ["option1", "option2", "option3"] }}
 onFilterChanged={(_rowId, value) => console.log(value)}
/>
```

## Used in table column filtering
``` ts
import { PluginColumns, MultipleSelector } from "@bunred/bunadmin"
export default ({ t }: PluginColumns): Column<Type>[] =>
[
  //...
  {
    title: t("Status"),
    field: "status",
    filterComponent: props => (
      <MultipleSelector {...props} />
    )
  }
]
```
*Refer to [plugin-example-blog/post/columns.tsx](https://github.com/bunred/bunadmin/blob/master/plugins/bunadmin-plugin-example-blog/post/columns.tsx)*

## Props
``` ts
type Props = {
  columnDef?: Column<any>
  onFilterChanged?: (rowId: string, value: any) => void
  valueToLowerCase?: boolean
  valueReplaceSpaces?: boolean
}
```
