# Drawer
### Used to build complex viewing / editing detail pages

## Default
<Drawer
  width="100%"
  height={225}
  direction="bottom"
  buttonTitle={"Default Drawer"}
>
  Container
</Drawer>

``` tsx
import { CommonDrawer as Drawer } from "@bunred/bunadmin"
//...
<Drawer
  width="100%"
  height={225}
  direction="bottom"
  buttonTitle={"Open Drawer"}
>
  Container
</Drawer>
```

## Props
``` ts
interface Props {
  width?: number | string
  height?: number | string
  direction?: "left" | "top" | "right" | "bottom"
  buttonTitle: string | JSX.Element
  buttonColor?: "inherit" | "default" | "primary" | "secondary" | undefined
  buttonVariant?: "text" | "outlined" | "contained"
  buttonSize?: "small" | "medium" | "large"
  buttonDisabled?: boolean
  buttonHidden?: boolean
  // switch toggleDrawer
  switchDrawer?: number
  // run func when open drawer
  onOpen?: (p: { contentRef: React.MutableRefObject<any | undefined> }) => void
  // run func when close drawer
  onClose?: (p: { contentRef: React.MutableRefObject<any | undefined> }) => void
  contentClassName?: string
  contentStyles?: CSS.Properties
  children: React.ReactNode
}
```
